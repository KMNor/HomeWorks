# Задание 3

#!/bin/bash
dirstat () {
        if [[ -d $name ]]; then
                fileq=`ls -la $name | grep "^-" | wc -l`
                files=`ls -la $name | grep "^-" | du -s | cut -f1`
                echo "$(($files/$fileq))"
        fi
}

name=/home/su
dirstat
