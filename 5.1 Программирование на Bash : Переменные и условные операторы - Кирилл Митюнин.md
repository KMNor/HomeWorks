# Задание 1
```
!/bin/bash
echo Введите директорию
read directory
if [[ -d "$directory" ]]; then
        echo "Директория $directory существует"
else  mkdir "$directory"
        echo "Директория $directory создана"
fi;
```

# Задание 2
```
#!/bin/bash
echo Введите первое число
read a
echo Введите второе число
read b
if [[ a -gt b ]]; then
        echo "$a-$b=$((a-b))"
elif [[ a -lt b ]]; then
        echo "$b-$a=$((b-a))"
else
        echo "$a*$b=$((a*b))"
fi;
```

```
#!/bin/bash
echo Введите имя файла с расширением
read filename
case $filename in
        *.jpg|*.gif|*.png)
                echo "Image"
        ;;
        *.mp3|*.wav)
                echo "Audio"
        ;;
        *.txt|*.doc)
                echo "Text"
        ;;
        *)
                echo "Unknown"
esac
```
